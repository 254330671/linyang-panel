function initChat(){var cfg=getApiConfig();var btn=document.getElementById('chat-send');if(cfg.key&&cfg.endpoint){btn.disabled=false;btn.textContent='发送';btn.style.opacity='1'}else{btn.disabled=true;btn.textContent='配置API';btn.style.opacity='0.5'}}
function sendChat(){var cfg=getApiConfig();if(!cfg.key)return;var input=document.getElementById('chat-input');var msg=input.value.trim();if(!msg)return;input.value='';chatHistory.push({role:'user',content:msg});
var ctx='[用户数据]\n';var tds=JSON.parse(localStorage.getItem('td')||'[]');if(tds.length){ctx+='待办：\n';tds.forEach(function(t){ctx+=(t.d?'✅':'⬜')+' '+t.t+(t.detail?' ('+t.detail+')':'')+'\n'})}var mds=JSON.parse(localStorage.getItem('mood')||'{}');if(mds.mood)ctx+='心情：'+mds.mood+' 能量：'+(mds.energy||'-')+'/5\n';var prof=JSON.parse(localStorage.getItem('profile')||'{}');if(prof.age)ctx+='档案：'+prof.age+'岁 '+(prof.city||'')+' '+(prof.jobs?prof.jobs.join('/'):'')+'\n';
renderChat();document.getElementById('chat-send').disabled=true;document.getElementById('chat-send').textContent='思考中...';
fetch(cfg.endpoint,{method:'POST',headers:{'Content-Type':'application/json','Authorization':'Bearer '+cfg.key},body:JSON.stringify({model:cfg.model||'gpt-4o-mini',messages:[{role:'system',content:ctx+'请根据以上用户数据回答，简洁实用，200字以内。'}].concat(chatHistory)})})
.then(function(r){return r.json()}).then(function(d){var reply=d.choices&&d.choices[0]&&d.choices[0].message?d.choices[0].message.content:'（无回复）';chatHistory.push({role:'assistant',content:reply});renderChat()})
.catch(function(e){chatHistory.push({role:'assistant',content:'⚠️ 请求失败：'+e.message});renderChat()})
.finally(function(){document.getElementById('chat-send').disabled=false;document.getElementById('chat-send').textContent='发送'})}
function renderChat(){var el=document.getElementById('chat-messages');el.innerHTML=chatHistory.map(function(m){var isU=m.role==='user';return '<div style="margin-bottom:8px;text-align:'+(isU?'right':'left')+'"><div style="display:inline-block;max-width:80%;padding:8px 12px;border-radius:12px;font-size:13px;line-height:1.5;background:'+(isU?'#2563eb':'#1e1e24')+';color:'+(isU?'#fff':'#e4e4e7')+'">'+m.content.replace(/</g,'&lt;').replace(/\n/g,'<br>')+'</div></div>'}).join('');el.scrollTop=el.scrollHeight}
initChat();

// === Profile ===
function openProfile(){var p=JSON.parse(localStorage.getItem('profile')||'{}');document.getElementById('prof-age').value=p.age||'24';document.getElementById('prof-city').value=p.city||'上海';document.querySelectorAll('#prof-gender .prof-tag').forEach(function(t){t.classList.toggle('set',t.dataset.v===p.gender)});document.getElementById('profile-modal').classList.add('open')}
function closeProfile(){var jobs=[];document.querySelectorAll('#prof-job .prof-tag.set').forEach(function(t){jobs.push(t.dataset.v)});localStorage.setItem('profile',JSON.stringify({age:document.getElementById('prof-age').value,city:document.getElementById('prof-city').value,gender:document.querySelector('#prof-gender .prof-tag.set')?document.querySelector('#prof-gender .prof-tag.set').dataset.v:'',jobs:jobs}));document.getElementById('profile-modal').classList.remove('open')}

if(window.innerWidth>600)return;
var homeEl=document.querySelector('.g5');
var sections=homeEl?homeEl.querySelectorAll('.sc'):[];
var cardSections=document.querySelectorAll('.c > .card');
// On mobile, show/hide by scrolling to section
var targets={'home':0,'life':1,'grow':2,'create':3,'health':4};
if(sections[targets[id]]){sections[targets[id]].scrollIntoView({behavior:'smooth'})}
document.querySelectorAll('.nav-btn').forEach(function(b){b.classList.remove('on')});
el.classList.add('on');
}
